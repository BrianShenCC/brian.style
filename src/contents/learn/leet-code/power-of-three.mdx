> Solution

``` typescript
function isPowerOfThree(n: number): boolean {
    if (n <= 0) {
        return false;
    } else if (n === 1) {
        return true;
    } else {
        let result = 3;
        for (let i = 1; i < 28; i++) {
            if (result === n) return true;
            result = result * 3;
        }
        return false;
    }
};
```
