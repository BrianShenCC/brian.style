> Solution

``` typescript
let memo: number[] = [];

function rob(nums: number[]): number {
    memo = Array(nums.length).fill(-1);
    return Math.max(getMaxByIndex(nums, 0), getMaxByIndex(nums, 1));
};

function getMaxByIndex(nums: number[], index: number): number {
    if (index >= nums.length) {
        return 0
    } else if (index === nums.length - 1 || index === nums.length - 2) {
        return nums[index]
    } else if (memo[index] !== -1) {
        return memo[index];
    } else {
        memo[index] = nums[index] + Math.max(getMaxByIndex(nums, index + 2), getMaxByIndex(nums, index + 3));
        return memo[index]
    }
}
```
