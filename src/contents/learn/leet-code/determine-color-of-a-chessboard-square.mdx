> Solution

``` typescript
function squareIsWhite(coordinates: string): boolean {
    const first = getNumber(coordinates[0]);
    const second = +coordinates[1] - 1;
    return ((first * 9 + second) % 2) !== 0
};

function getNumber(c: string) {
    return c.charCodeAt(0) - 97;
}
```
