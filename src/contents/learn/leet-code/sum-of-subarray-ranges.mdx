> Solution

``` typescript
function subArrayRanges(nums: number[]): number {
    let sum = 0;
    for (let i = 0, length = nums.length; i < length - 1; i++) {
        let max = nums[i], min = nums[i];
        for (let j = i + 1; j < length; j++) {
            if (nums[j] > max) max = nums[j]
            if (nums[j] < min) min = nums[j]
            sum += max - min;
        }
    }
    return sum;
};
```
