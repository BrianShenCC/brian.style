> Solution

``` typescript
function numberOfLines(widths: number[], s: string): number[] {
    let result = 1;
    let lastLineWidth = 0;
    for (let i = 0; i < s.length; i++) {
        let width = widths[s.charCodeAt(i) - 97];
        if (width + lastLineWidth > 100) {
            result++;
            lastLineWidth = width;
        } else {
            lastLineWidth += width;
        }
    }
    return [result, lastLineWidth];
};
```
