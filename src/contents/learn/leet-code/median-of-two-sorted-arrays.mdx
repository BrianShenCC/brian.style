function findMedianSortedArrays(nums1: number[], nums2: number[]): number {
    // find median if it's odd n/2+1
    // if Even, find (n/2-1 + n/2+1)/2
    if ((nums1.length + nums2.length) % 2) {
        const mid = Math.floor((nums1.length + nums2.length) / 2);
        return findNthNumber(nums1, nums2, mid)
    } else {
        const midleft = ((nums1.length + nums2.length) / 2) - 1;
        return (findNthNumber(nums1, nums2, midleft) + findNthNumber(nums1, nums2, midleft + 1)) / 2
    }
};

function findNthNumber(nums1: number[], nums2: number[], n: number): number {
    let i = 0, j = 0;
    let result;
    while (i + j <= n) {
        if (nums1[i] === undefined) {
            result = nums2[j]
            j++
        } else if (nums2[j] === undefined) {
            result = nums1[i]
            i++
        } else if (nums1[i] < nums2[j]) {
            result = nums1[i]
            i++
        } else {
            result = nums2[j]
            j++
        }
    }
    return result;
}
