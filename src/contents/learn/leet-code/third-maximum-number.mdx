> Solution

``` typescript
function thirdMax(nums: number[]): number {
    let ignoreNums: number[] = [];
    let max = getMax(nums, ignoreNums);
    let maxCount = 0;
    while (maxCount < 2) {
        ignoreNums.push(max);
        max = getMax(nums, ignoreNums);
        maxCount++;
    }
    return max === -Infinity ? ignoreNums[0] : max;
};

function getMax(nums: number[], ignoreNums: number[]): number {
    return nums.reduce((max, num) => {
        if (ignoreNums.includes(num)) return max;
        return num > max ? num : max;
    }, -Infinity);
}
```
