> Solution

``` typescript
function removeElement(nums: number[], val: number): number {
    const filtered = nums.filter(num => num !== val);
    nums.forEach((n, index) => {
        nums[index] = filtered[index] ?? '_' as unknown as number;
    })
    return filtered.length;
};
```
