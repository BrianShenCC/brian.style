> Solution

``` typescript
let memo: number[] = [];

function minCostClimbingStairs(cost: number[]): number {
    memo = Array(cost.length).fill(-1);
    return Math.min(getMinByIndex(cost, 0), getMinByIndex(cost, 1));
};

function getMinByIndex(cost: number[], index: number): number {
    if (index === cost.length - 1 || index === cost.length - 2) {
        return cost[index]
    } else if (memo[index] !== -1) {
        return memo[index];
    } else {
        memo[index] = cost[index] + Math.min(getMinByIndex(cost, index + 1), getMinByIndex(cost, index + 2));
        return memo[index]
    }
}
```
