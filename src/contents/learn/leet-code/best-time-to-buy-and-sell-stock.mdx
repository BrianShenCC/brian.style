> Solution

``` typescript
function maxProfit(prices: number[]): number {
    let profit = 0;
    let buy = Infinity;
    for (let i = 0; i < prices.length; i++) {
        buy = Math.min(prices[i], buy);
        profit = Math.max(profit, prices[i] - buy);
    }
    return profit;
};
```
