https://leetcode.com/problems/reverse-string/

> Solution

``` typescript
/**
 Do not return anything, modify s in-place instead.
 */
function reverseString(s: string[]): void {
    let i = s.length - 1;
    for (let j = 0; j <= i; j++) {
        let val = s[i];
        s[i] = s[j]
        s[j] = val
        i--
    }
};
```
