``` typescript
function isPowerOfTwo(n: number): boolean {
    if (n <= 0) {
        return false;
    } else if (n === 1) {
        return true;
    } else {
        let result = 2;
        for (let i = 1; i < 31; i++) {
            if (result === n) return true;
            result = result << 1;
        }
        return false;
    }
};
```
