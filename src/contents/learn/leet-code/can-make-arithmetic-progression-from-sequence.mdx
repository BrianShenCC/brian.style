> Solution

``` typescript
function canMakeArithmeticProgression(nums: number[]): boolean {
    let count;
    for (let i = 0; i < nums.length; i++) {
        let max = i;
        for (let j = i; j < nums.length; j++) {
            if (nums[j] > nums[max]) {
                max = j;
            }
        }
        let tmp = nums[i];
        nums[i] = nums[max];
        nums[max] = tmp;
        if (i == 0) {
            continue;
        } else if (i === 1) {
            count = nums[i - 1] - nums[i];
        } else if (nums[i - 1] - nums[i] !== count) {
            return false;
        }

    }
    return true
};
```
