> Solution

``` typescript
/**
 Do not return anything, modify nums in-place instead.
 */
function moveZeroes(nums: number[]): void {
    let length = nums.length;
    let fast = 0, slow = 0;

    while (fast < length) {
        if (nums[fast] !== 0) {
            nums[slow] = nums[fast];
            slow++;
        }
        fast++;
    }
    while (slow < length) {
        nums[slow] = 0;
        slow++
    }
};
```
