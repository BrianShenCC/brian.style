``` typescript
function isPowerOfFour(n: number): boolean {
    if (n <= 0) {
        return false;
    } else if (n === 1) {
        return true;
    } else {
        let result = 4;
        for (let i = 1; i < 16; i++) {
            if (result === n) return true;
            result = result << 2;
        }
        return false;
    }
};
```
