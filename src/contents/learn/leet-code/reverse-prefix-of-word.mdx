> Solution

``` typescript
function reversePrefix(word: string, ch: string): string {
    let i = word.indexOf(ch);
    if (i === -1) return word;
    const list = [...word];
    const reverseList = list.splice(0, i + 1);
    for (let j = 0; j <= i; j++) {
        let val = reverseList[i];
        reverseList[i] = reverseList[j]
        reverseList[j] = val
        i--
    }
    return reverseList.join("") + list.join("");
};
```
