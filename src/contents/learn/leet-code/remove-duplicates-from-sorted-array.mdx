> Solution

``` typescript
function removeDuplicates(nums: number[]): number {
  const removed = Array.from(new Set(nums))
  for(let i = 0; i < removed.length; i++) {
    nums[i] = removed[i] ?? '_' as unknown as number;
  }
  return removed.length;
};
```
