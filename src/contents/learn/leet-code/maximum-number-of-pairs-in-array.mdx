https://leetcode.com/problems/maximum-number-of-pairs-in-array/

> Solution

``` typescript
function numberOfPairs(nums: number[]): number[] {
    let numberLeft = 0;
    let numberPairs = 0;
    const removedIndex = [];
    nums.forEach((num, index) => {
        if (removedIndex.includes(index)) {
            return;
        }
        if (index === nums.length - 1) {
            numberLeft += 1;
            return;
        }
        const pairIndex = nums.indexOf(num, index + 1)
        if (pairIndex === -1) {
            numberLeft += 1;
            return;
        } else {
            numberPairs += 1;
            removedIndex.push(pairIndex);
        }
    });
    return [numberPairs, numberLeft];
};
```
