> Solution

``` typescript
function maximizeSum(nums: number[], k: number): number {
    const max = getMax(nums);
    const maxKget = max + k - 1;
    //   const maxK = max + k;
    //   const index = nums.findLastIndex((num) => num === max);
    //   nums[index] = maxK;
    return (max + maxKget) / 2 * k;
};

function getMax(nums: number[]) {
    return nums.reduce((acc, curr) => Math.max(acc, curr), -Infinity);
}
```
