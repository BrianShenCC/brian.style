``` typescript
function twoOutOfThree(nums1: number[], nums2: number[], nums3: number[]): number[] {
    const array1 = intersection(nums1, nums2);
    const array2 = intersection(nums2, nums3);
    const array3 = intersection(nums3, nums1);
    return Array.from(new Set(array1.concat(array2,array3)))
};

function intersection(nums1: number[], nums2: number[]): number[] {
    const set = new Set(nums1);
    const filtered = Array.from(set);
    return filtered.filter((num) => nums2.includes(num));
};
```
