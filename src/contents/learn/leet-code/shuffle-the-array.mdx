> Solution

``` typescript
function shuffle(nums: number[], n: number): number[] {
    const result = new Array(nums.length);
    const half = nums.splice(n, n);
    for (let i = 0; i < n; i++) {
        result[2 * i] = nums[i];
        result[2 * i + 1] = half[i];
    }
    return result;
};
```
