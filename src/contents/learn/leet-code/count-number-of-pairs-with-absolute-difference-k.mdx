> Solution

``` typescript
function countKDifference(nums: number[], k: number): number {
    if (nums.length === 0) {
        return 0
    }
    let result = 0;
    const l = nums.length;

    for (let i = 0; i <= l; i++) {
        let first = nums[i];
        for (let j = i + 1; j <= l; j++) {
            const second = nums[j];
            if (second - first === k || first - second === k) {
                result++;
            }
        }
    }
    return result;
};
```
