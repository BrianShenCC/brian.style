> Solution

``` typescript
function twoSum(nums: number[], target: number): number[] {
    const object = {};
    const length = nums.length;
    for (let i = 0; i < length; i++) {
        const num = nums[i];
        if (object[target - num] !== undefined) {
            return [object[target - num], i]
        }
        object[num] = i;
    }
    throw Error('result not found')
};
```
