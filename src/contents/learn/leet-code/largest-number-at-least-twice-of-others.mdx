> Solution

``` typescript
function dominantIndex(nums: number[]): number {
    // find 2 largest number;
    let index = -1;
    for (let i = 0; i < 2; i++) {
        let max = i;
        for (let j = i; j < nums.length; j++) {
            if (nums[j] > nums[max]) {
                max = j;
            }
        }
        let tmp = nums[i];
        nums[i] = nums[max];
        nums[max] = tmp;
        // mark the lagest index;
        if (i === 0) {
            index = max;
        }
    }
    if (nums[0] >= nums[1] * 2) {
        return index;
    }
    return -1
};
```
