> Solution

``` typescript
/**
 Do not return anything, modify nums in-place instead.
 */
// using selection sort
function sortColors(nums: number[]): void {
    for (let i = 0; i < nums.length; i++) {
        let min = i;
        for (let j = i; j < nums.length; j++) {
            if (nums[j] < nums[min]) {
                min = j;
            }
        }
        let tmp = nums[i];
        nums[i] = nums[min];
        nums[min] = tmp;
    }
};
```
