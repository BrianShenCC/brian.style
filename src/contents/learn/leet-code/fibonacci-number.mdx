> Solution

``` typescript
function fib(n: number): number {
    if (n === 0) return 0;
    if (n === 1) return 1;
    let results = [0, 1];
    let result = 0;
    for (let i = 2; i <= n; i++) {
        result = results[0] + results[1];
        results.push(result);
        results.shift();
    }
    return result;
};
```
