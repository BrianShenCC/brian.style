> Solution

``` typescript
function differenceOfSum(nums: number[]): number {
    const sum = nums.reduce((acc, num) => acc + num, 0);
    const sum2 = nums.reduce((acc, num) => acc + count(num), 0);
    return sum - sum2;
};
const count = (num: number) => {
    let result = 0;
    let temp = num;
    while (temp / 10 > 0) {
        result += temp % 10;
        temp = (temp - temp % 10) / 10;
    }
    return result;
}
```
