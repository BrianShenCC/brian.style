``` typescript
function reverseWords(s: string): string {
    s = s.trim().replace(/[ ]+/g,' ');
    const list = s.split(' ').reverse();
    let result = '';
    let blankFlag = false;
    let wordIndex = 0;
    for (let i = 0, len = s.length; i < len; i++) {
        blankFlag = (s[i] === ' ')
        if (blankFlag) {
            result += ' ';
        }
        if (!blankFlag) {
            if (s[i + 1] === undefined) {
                result += list[wordIndex];
            }
            if (s[i + 1] !== ' ') {
                continue;
            }
            if (s[i + 1] === ' ') {
                result += list[wordIndex];
                wordIndex++;
            }
        }
    }
    return result
};
```
