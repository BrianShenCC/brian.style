> Solution

``` typescript
function numberOfBeams(bank: string[]): number {
  const filterNoDevices = bank.filter(item => item.includes('1'))
  let lastRowDevicesNums = 0;
  return filterNoDevices.reduce((prev, next) => {
      const currentDevicesNums = countDevices(next);
      const beams = currentDevicesNums * lastRowDevicesNums;
      lastRowDevicesNums = currentDevicesNums;
      return prev + beams;
  }, 0)
};
const countDevices = (devicesString: string) => {
  let result = 0
  for (let i = 0, length = devicesString.length; i < length; i++) {
      if (devicesString[i] === '1') result += 1;
  }
  return result;
}
```
